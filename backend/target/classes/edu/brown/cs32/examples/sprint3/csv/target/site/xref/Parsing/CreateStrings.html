3   import Exceptions.FactoryFailureException;
4   import java.util.List;
5   
6   /**
7    * This class implements the CreatorFromRow interface and converts the row into a List of strings.
8    */
9   public class CreateStrings implements CreatorFromRow<StringBuilder> {
10    @Override
11    public StringBuilder create(List<String> row) throws FactoryFailureException {
12      StringBuilder concatenatedString = new StringBuilder();
13  
14      for (String string : row) {
15        concatenatedString.append(string).append(" ");
16      }
17      return concatenatedString;
18    }
19  }
